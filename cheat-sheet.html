データの抽出

select 列1, 列2, ... from テーブル名 where 条件 limit (N番目), M番目

例

select * from students where pref='東京都';

select id, zip, pref from students where not id between 10010 and 10019 and pref like '%県';

select id, tel from students where tel is not null and id < 10010;

select * from students limit 3;

select * from students where class_id=3 limit 2, 4;

select id, family_name, given_name, pref from students where tel=090 limit 2;

select id, family_name, given_name, pref from students where id between 10030 and 10039 and (pref="千葉県" or pref="茨城県" or pref="群馬県");

関係データベースとSQL

テーブルの作成・削除

create table running_club(
id integer primary key autoincrement,
student_id integer not null unique,
team text,
foreign key(student_id) references students(id));

drop table running_club;

データの登録・変更・削除

insert into running_club values(1, 10001, '短距離');

insert into running_club values(2, 10003, '長距離');

insert into running_club(student_id, team) values(10005, '短距離');

update running_club set team='長距離' where id=1;

delete from running_club where student_id=10005;
